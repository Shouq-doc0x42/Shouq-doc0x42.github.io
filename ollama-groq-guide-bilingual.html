# Create a .env file (NEVER commit this)
# GROQ_API_KEY=gsk_your_key_here

import os
from groq import Groq
from dotenv import load_dotenv

load_dotenv()  # Load .env file

client = Groq(
    api_key=os.getenv("GROQ_API_KEY")
)

response = client.chat.completions.create(
    model="llama3-8b-8192",
    messages=[
        {
            "role": "user",
            "content": "What are common indicators of a phishing attack?"
        }
    ]
)

print(response.choices[0].message.content)

Python · Universal AI Query Function
def query_ai(prompt, use_local=True):
    """
    Universal: send prompt to Ollama (local) or Groq (cloud).
    use_local=True  → Ollama  (private, free, offline-capable)
    use_local=False → Groq    (fast, cloud, larger context)
    """
    if use_local:
        resp = requests.post(
            "http://localhost:11434/api/generate",
            json={"model": "llama3", "prompt": prompt, "stream": False}
        )
        return resp.json()["response"]
    else:
        client = Groq(api_key=os.getenv("GROQ_API_KEY"))
        resp = client.chat.completions.create(
            model="llama3-70b-8192",
            messages=[{"role": "user", "content": prompt}]
        )
        return resp.choices[0].message.content

result = query_ai("Analyze suspicious activity", use_local=True)
result = query_ai("Generate incident report",    use_local=False)
